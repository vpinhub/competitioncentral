function copyToClipboard(text, btnElement) { const originalText = btnElement.innerText; const showSuccess = () => { btnElement.innerText = 'Copied!'; setTimeout(() => { btnElement.innerText = originalText; }, 2000); }; const showError = () => { btnElement.innerText = 'Error! Check Console.'; setTimeout(() => { btnElement.innerText = originalText; }, 2000); }; // Modern API (Requires HTTPS and secure iframe) if (navigator.clipboard && window.isSecureContext) { navigator.clipboard.writeText(text) .then(showSuccess) .catch(err => { console.warn('Modern clipboard failed, trying fallback.', err); fallbackCopy(text); }); } // Fallback for HTTP or restrictive iframes else { fallbackCopy(text); } function fallbackCopy(textToCopy) { try { const textArea = document.createElement("textarea"); textArea.value = textToCopy; // Keep it completely hidden from the user textArea.style.position = "fixed"; textArea.style.left = "-9999px"; textArea.style.top = "-9999px"; document.body.appendChild(textArea); textArea.select(); const successful = document.execCommand("copy"); textArea.remove(); if (successful) { showSuccess(); } else { showError(); } } catch (err) { console.error("Fallback copy failed: ", err); showError(); } } }
